SELECT COLUMN_NAME, DATA_TYPE, CHARACTER_MAXIMUM_LENGTH, IS_NULLABLE
FROM INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_NAME = 'TempContractTypes' AND COLUMN_NAME = 'ContractTypes';


SELECT i.name AS IndexName, c.name AS ConstraintName
FROM sys.indexes i
LEFT JOIN sys.check_constraints c ON c.parent_object_id = i.object_id
WHERE i.object_id = OBJECT_ID('RSSDynamicMenusLookup.dbo.TempContractTypes');


ALTER TABLE RSSDynamicMenusLookup.dbo.TempContractTypes 
ALTER COLUMN ContractTypes NVARCHAR(MAX);
